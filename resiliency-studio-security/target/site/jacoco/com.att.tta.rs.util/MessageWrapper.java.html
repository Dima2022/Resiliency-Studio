package com.att.tta.rs.util;

import java.util.Arrays;
import java.util.List;

import org.springframework.http.HttpStatus;

/**
 * This class provides wrapper to encapsulate the API response message with http status.  
 * @author ak983d
 *
 */
public class MessageWrapper {

	private HttpStatus status;
	private String statusMessage;
	private List<String> statusDesc;

	public MessageWrapper() {
		super();
	}

	public MessageWrapper(final HttpStatus status, final String statusMessage, final List<String> statusDesc) {
		super();
		this.status = status;
		this.statusMessage = statusMessage;
		this.statusDesc = statusDesc;
	}

	public MessageWrapper(final HttpStatus status, final String statusMessage, final String statusDesc) {
		super();
		this.status = status;
		this.statusMessage = statusMessage;
		this.statusDesc = Arrays.asList(statusDesc);
	}

	public HttpStatus getStatus() {
		return status;
	}

	public void setStatus(HttpStatus status) {
		this.status = status;
	}

	public String getStatusMessage() {
		return statusMessage;
	}

	public void setStatusMessage(String statusMessage) {
		this.statusMessage = statusMessage;
	}

	public List<String> getStatusDesc() {
		return statusDesc;
	}

	public void setStatusDesc(List<String> statusDesc) {
		this.statusDesc = statusDesc;
	}	

}
