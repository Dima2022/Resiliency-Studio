package com.att.tta.rs.model;

import com.fasterxml.jackson.annotation.JsonValue;

/**
 * This provides Enum constants for Job Execution Status.
 * 
 * @author ak983d
 *
 */
public enum EventStatus {
	SUBMITTED("Submitted"), SCHEDULED("Scheduled"), INPROGRESS("In Progress"), SUCCESS("Success"), FAILED(
			"Failed"), COMPLETED("Completed"), REJECTED("Rejected");

	private String evntSts;

	EventStatus(String eventStatus) {
		this.evntSts = eventStatus;
	}

	@JsonValue
	public String evntSts() {
		return this.evntSts;
	}

	public static EventStatus fromString(String text) {
		if (text != null) {
			for (EventStatus b : EventStatus.values()) {
				if (text.equalsIgnoreCase(b.evntSts)) {
					return b;
				}
			}
		}
		return null;
	}
}
