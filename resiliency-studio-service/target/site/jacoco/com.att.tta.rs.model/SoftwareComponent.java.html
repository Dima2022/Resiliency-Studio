package com.att.tta.rs.model;

import java.io.Serializable;

/**
 * This class is the representation of 'Software Component' Data Model (DTO) in the Elastic Search.
 * 
 * @author ak983d
 */

public class SoftwareComponent implements Serializable {

	private static final long serialVersionUID = 1L;

	/** The server component name . */
	private String serverComponentName;

	/** The server component process name . */
	private String processName;

	/**
	 * @return the serverComponentName
	 */
	public String getServerComponentName() {
		return serverComponentName;
	}

	/**
	 * @param serverComponentName
	 *            the serverComponentName to set
	 */
	public void setServerComponentName(String serverComponentName) {
		this.serverComponentName = serverComponentName;
	}

	/**
	 * @return the processName
	 */
	public String getProcessName() {
		return processName;
	}

	/**
	 * @param processName
	 *            the processName to set
	 */
	public void setProcessName(String processName) {
		this.processName = processName;
	}

}
